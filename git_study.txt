git clone ''
#가져오고 싶은 계정의 레파지토리를 클론 해 온다.

git status
# 로컬에서 수정한 파일 이 있는지 확인

git add hello.py
# git에 업로드 할 파일을 add
# git add -A > 작업한 모든 파일을 올리기 위한 선행 단계

git commit -m ""
# git에 업로드 할때 코멘트를 적는다.
# 실제 add 에서 작업한 파일을 실제 적용하기 전 단계

git push origin [master, Develop....등]
# 최종적으로 git에 반영할 것
# git에 최종적으로 업데이트시 원하는 branch에 업데이트 할때

git branch [name]
# 브랜치를 만든다.

git checkout [name]
# 현재 브랜치를 이동
